Build System (Colcon)

Master ROS2's build tool, package structure, and dependency management.

Colcon CMake

What / Why / When

What

Colcon is ROS2's build tool, succeeding catkin from ROS1. It builds workspace packages independently with correct dependency order. Each package has package.xml (declares dependencies, metadata) and build files (CMakeLists.txt for C++, setup.py for Python). Colcon supports ament_cmake (C++), ament_python (Python), and cmake (plain CMake). Workspace structure: src/ (source), build/ (intermediates), install/ (final binaries), log/ (build logs). Colcon provides parallel builds, symlink installs (fast iteration), and isolated builds (prevent cross-contamination).

Why

Manual compilation of multi-package workspaces is error-prone - dependency order matters. Colcon automates topological sorting: builds dependencies first. Incremental builds are fast - only modified packages rebuild. --symlink-install avoids copying Python files, enabling instant code changes. Colcon's isolation prevents "it works on my machine" issues - each package builds in clean environment. Build flags propagate correctly (e.g., CMAKE_BUILD_TYPE). Colcon integrates with CI/CD for reproducible builds.

When

Use colcon build for all multi-package workspaces. Single packages can use colcon build --packages-select my_pkg. Use --symlink-install for Python development (no reinstall needed). Build types: Debug (symbols, no optimization) for development, Release (optimized) for production, RelWithDebInfo (optimized + symbols) for profiling. Clean builds (rm -rf build install) resolve mysterious errors. Always source install/setup.bash after building to update environment.

10-Minute Drill

Basic Workspace Structure

# Create workspace
mkdir -p ~/ros2_ws/src
cd ~/ros2_ws

# Create package (C++)
cd src
ros2 pkg create my_robot_pkg \
  --build-type ament_cmake \
  --dependencies rclcpp std_msgs geometry_msgs

# Create package (Python)
ros2 pkg create my_py_pkg \
  --build-type ____  \  # ament_python
  --dependencies rclpy std_msgs

# Workspace structure
# ros2_ws/
#   src/
#     my_robot_pkg/
#       CMakeLists.txt
#       package.xml
#       include/
#       src/
#     my_py_pkg/
#       package.xml
#       setup.py
#       my_py_pkg/

Essential Colcon Commands

Command Purpose Use Case
colcon build Build all packages Fresh workspace, full rebuild
colcon build --packages-select my_pkg Build one package Fast iteration on single package
colcon build --symlink-install Symlink Python files Python dev (no reinstall needed)
colcon build --cmake-args -DCMAKE_BUILD_TYPE=Release Optimized build Production deployment
colcon test Run all tests CI/CD verification
colcon test-result --verbose Show test output Debug test failures

Package.xml Structure

<?xml version="1.0"?>
<package format="3">
  <name>my_robot_pkg</name>
  <version>0.1.0</version>
  <description>My robot navigation package</description>
  <maintainer email="you@example.com">Your Name</maintainer>
  <license>Apache-2.0</license>

  <!-- Build tool -->
  <buildtool_depend>____</buildtool_depend>  <!-- ament_cmake -->

  <!-- Build and runtime dependencies -->
  <depend>rclcpp</depend>
  <depend>std_msgs</depend>
  <depend>geometry_msgs</depend>

  <!-- Build-only dependencies -->
  <build_depend>rosidl_default_generators</build_depend>
  
  <!-- Runtime-only dependencies -->
  <exec_depend>rosidl_default_runtime</exec_depend>

  <!-- Test dependencies -->
  <test_depend>ament_lint_auto</test_depend>
  <test_depend>ament_lint_common</test_depend>

  <export>
    <build_type>ament_cmake</build_type>
  </export>
</package>

CMakeLists.txt (ament_cmake)

cmake_minimum_required(VERSION 3.8)
project(my_robot_pkg)

# Compiler settings
if(CMAKE_COMPILER_IS_GNUCXX OR CMAKE_CXX_COMPILER_ID MATCHES "Clang")
  add_compile_options(-Wall -Wextra -Wpedantic)
endif()

# Find dependencies
find_package(ament_cmake REQUIRED)
find_package(rclcpp REQUIRED)
find_package(std_msgs REQUIRED)
find_package(geometry_msgs REQUIRED)

# Add executable
add_executable(my_node src/my_node.cpp)
ament_target_dependencies(my_node
  rclcpp
  std_msgs
  ____  # geometry_msgs
)

# Install executable
install(TARGETS my_node
  DESTINATION lib/${PROJECT_NAME}
)

# Install launch files
install(DIRECTORY launch
  DESTINATION share/${PROJECT_NAME}
)

# Install config files
install(DIRECTORY config
  DESTINATION share/${PROJECT_NAME}
)

# Testing
if(BUILD_TESTING)
  find_package(ament_lint_auto REQUIRED)
  ament_lint_auto_find_test_dependencies()
endif()

Concept ↔ Code (Build Process)

Colcon Build Workflow

# 1. Dependency resolution
# Colcon reads package.xml in src/
# Builds topological sort: if pkg_b depends on pkg_a, build pkg_a first

# 2. Configure phase (CMake)
# Creates build/my_pkg/
# Runs: cmake -DCMAKE_INSTALL_PREFIX=install ...

# 3. Build phase
# Compiles C++ code: make -j$(nproc)
# Or for Python: copies files

# 4. Install phase
# Copies binaries to install/my_pkg/lib/
# Copies headers to install/my_pkg/include/
# Copies config/launch to install/my_pkg/share/

# 5. Environment setup
# Creates install/setup.bash with:
# - PATH: adds install/my_pkg/lib/my_pkg
# - LD_LIBRARY_PATH: adds install/my_pkg/lib
# - AMENT_PREFIX_PATH: adds install/my_pkg

Custom CMake Targets

# Add library (for reusable code)
add_library(my_lib src/my_library.cpp)
ament_target_dependencies(my_lib rclcpp)

# Link executable to library
add_executable(my_node src/my_node.cpp)
target_link_libraries(my_node ____)  # my_lib

# Install library and headers
install(TARGETS my_lib
  ARCHIVE DESTINATION lib
  LIBRARY DESTINATION lib
  RUNTIME DESTINATION bin
)

install(DIRECTORY include/
  DESTINATION include
)

# Export library for other packages
ament_export_targets(export_my_lib HAS_LIBRARY_TARGET)
install(TARGETS my_lib
  EXPORT export_my_lib
  LIBRARY DESTINATION lib
  ARCHIVE DESTINATION lib
  RUNTIME DESTINATION bin
  INCLUDES DESTINATION include
)

Python Setup.py (ament_python)

from setuptools import setup

package_name = 'my_py_pkg'

setup(
    name=package_name,
    version='0.1.0',
    packages=[package_name],
    data_files=[
        # Install package.xml
        ('share/ament_index/resource_index/packages',
            ['resource/' + package_name]),
        ('share/' + package_name, ['package.xml']),
        
        # Install launch files
        ('share/' + package_name + '/launch',
            ['launch/my_launch.py']),
        
        # Install config files
        ('share/' + package_name + '/config',
            ['____/my_config.yaml']),  # config
    ],
    install_requires=['setuptools'],
    zip_safe=True,
    maintainer='Your Name',
    maintainer_email='you@example.com',
    description='My Python package',
    license='Apache-2.0',
    tests_require=['pytest'],
    entry_points={
        'console_scripts': [
            'my_node = my_py_pkg.my_node:main',
        ],
    },
)

Parameters & Config

Build Type Selection

# Debug build (default): no optimization, debug symbols
colcon build --cmake-args -DCMAKE_BUILD_TYPE=Debug

# Release build: full optimization, no symbols
colcon build --cmake-args -DCMAKE_BUILD_TYPE=____  # Release

# RelWithDebInfo: optimization + symbols (for profiling)
colcon build --cmake-args -DCMAKE_BUILD_TYPE=RelWithDebInfo

# Check current build type
cat build/my_pkg/CMakeCache.txt | grep CMAKE_BUILD_TYPE

Parallel Build Control

# Limit parallel jobs (reduce RAM usage)
colcon build --parallel-workers 2

# Or set environment variable
export MAKEFLAGS="-j2"
colcon build

# Default: uses all CPU cores $(nproc)

Selective Building

# Build single package
colcon build --packages-select my_pkg

# Build package and dependencies
colcon build --packages-up-to my_pkg

# Skip specific packages
colcon build --packages-skip test_pkg

# Continue on error (build as many as possible)
colcon build --continue-on-error

Colcon Defaults File

# Save default arguments: ~/.colcon/defaults.yaml
build:
  symlink-install: true
  cmake-args:
    - "-DCMAKE_BUILD_TYPE=____"  # RelWithDebInfo
  parallel-workers: 4

test:
  pytest-with-coverage: true

Verify

Build Verification

# Check build succeeded
echo $?
# Expected: 0 (success)

# Verify executable installed
ls install/my_pkg/lib/my_pkg/
# Should see: my_node

# Verify shared libraries
ls install/my_pkg/lib/*.so
# Should see: libmy_lib.so

# Check if package visible to ROS2
source install/setup.bash
ros2 pkg list | grep my_pkg
# Should see: my_pkg

# Run installed executable
ros2 run my_pkg my_node
# Should start without errors

Build Logs

# Check build logs
cat log/latest_build/my_pkg/stdout.log
cat log/latest_build/my_pkg/stderr.log

# Find all errors in workspace
grep -r "error:" log/latest_build/

# Check warnings
grep -r "warning:" log/latest_build/ | wc -l

Dependency Verification

# List package dependencies
colcon list --packages-select my_pkg

# Check if dependencies met
rosdep check --from-paths src --ignore-src

# Install missing dependencies
rosdep install --from-paths src --ignore-src -y
Pro Tip:

Use colcon build --event-handlers console_direct+ for immediate output during build. Helpful for debugging stuck builds or seeing progress in real-time.

Common Failures & Fix Patterns

Failure 1: Package Not Found

Symptom: CMake Error: Could not find a package configuration file provided by "xxx"

Root Cause: Missing dependency or dependency not in workspace.

Fix: Install dependency or build it:

# Check if system package available
apt search ros-humble-xxx

# Install if available
sudo apt install ros-humble-____  # package-name

# Or add to package.xml
<depend>xxx</depend>

# And rebuild
colcon build --packages-select my_pkg

Failure 2: Undefined Reference (Linker Error)

Symptom: undefined reference to 'MyClass::myFunction()'

Root Cause: Missing library link in CMakeLists.txt.

Fix: Link library:

# If using external library
find_package(my_lib REQUIRED)
ament_target_dependencies(my_node my_lib)

# If using library in same package
target_link_libraries(my_node ____)  # my_internal_lib

Failure 3: Python Module Not Found

Symptom: ModuleNotFoundError: No module named 'my_py_pkg'

Root Cause: Forgot to source workspace or missing --symlink-install.

Fix: Source and rebuild with symlink:

# Rebuild with symlink
colcon build --symlink-install --packages-select my_py_pkg

# Source workspace
source install/setup.bash

# Verify package visible
python3 -c "import my_py_pkg"
# No error = success

Failure 4: Old Build Artifacts

Symptom: Mysterious build errors, CMake finds wrong files.

Root Cause: Stale build/install directories.

Fix: Clean rebuild:

# Remove build artifacts
rm -rf build/ install/ log/

# Fresh build
colcon build

Failure Injection Drill

Exercise: Missing Dependency

Remove dependency from CMakeLists.txt but keep code using it:

# CMakeLists.txt - intentional error
find_package(ament_cmake REQUIRED)
find_package(rclcpp REQUIRED)
# find_package(geometry_msgs REQUIRED)  <!-- COMMENTED OUT -->

add_executable(my_node src/my_node.cpp)
ament_target_dependencies(my_node
  rclcpp
  # geometry_msgs  <!-- MISSING -->
)
// src/my_node.cpp - uses geometry_msgs
#include "rclcpp/rclcpp.hpp"
#include "geometry_msgs/msg/twist.hpp"  // WILL FAIL

int main() {
    geometry_msgs::msg::Twist cmd;  // Error: type not found
    return 0;
}

Observe:

colcon build --packages-select my_pkg
# Error: 'geometry_msgs/msg/twist.hpp' file not found
# Or: 'geometry_msgs' is not a member of...

# Build fails at compile time

Detection: Check stderr log:

cat log/latest_build/my_pkg/stderr.log
# Shows missing header file

Recovery:

# Add missing dependency
find_package(geometry_msgs REQUIRED)
ament_target_dependencies(my_node rclcpp ____) # geometry_msgs
Warning:

Always keep package.xml and CMakeLists.txt in sync. Dependency in code must be in both files: <depend> in package.xml AND find_package() in CMakeLists.txt.
